// PATTERN 1: Direct Client (Community Tier, CLI, Simple Apps)
import { OuroCClient } from '@ouroc/sdk'

const client = new OuroCClient("your-canister-id", "devnet")

// Direct subscription creation
async function createSubscription() {
  try {
    const subscriptionId = await client.createSubscription({
      subscription_id: "unique-subscription-id",
      solana_contract_address: "your-solana-contract",
      subscriber_address: "user-wallet-address",
      merchant_address: "merchant-wallet-address",
      payment_token_mint: "USDC-mint-address",
      amount: 29000000, // 29 USDC in micro-units
      intervalSeconds: 2592000, // 30 days
      api_key: "your-api-key"
    })
    console.log('✅ Subscription created:', subscriptionId)
  } catch (error) {
    console.error('❌ Subscription failed:', error)
  }
}

// PATTERN 2: React Provider (Full-featured Apps)
import { OuroCProvider, SubscriptionCard, useSubscription } from '@ouroc/sdk'

function App() {
  return (
    <OuroCProvider canisterId="your-canister-id" network="devnet">
      <SubscriptionComponent />
    </OuroCProvider>
  )
}

function SubscriptionComponent() {
  const { createSubscription } = useSubscription()

  const handleSubscribe = async (plan) => {
    try {
      const subscriptionId = await createSubscription({
        subscription_id: "unique-subscription-id",
        solana_contract_address: "your-solana-contract",
        // subscriber_address auto-filled from connected wallet
        merchant_address: "merchant-wallet-address",
        payment_token_mint: "USDC-mint-address",
        amount: 29000000, // 29 USDC in micro-units
        intervalSeconds: 2592000, // 30 days
        api_key: "your-api-key"
      })
      console.log('✅ Subscription created:', subscriptionId)
    } catch (error) {
      console.error('❌ Subscription failed:', error)
    }
  }

  return (
    <SubscriptionCard
      planName="Pro"
      price={29} // 29 USDC (whole tokens)
      interval="monthly"
      features={['AI Features', 'Priority Support']}
      onSubscribe={handleSubscribe}
    />
  )
}
